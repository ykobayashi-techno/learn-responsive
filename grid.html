/* grid-template-areasで適当なラベルを付けてレイアウト　空の部分には.を使える */
.freeGgridContainer {
    display: grid;
    grid-template-areas:    "es es es lb lb lb lb"
                            "ib .. .. .. .. .. .."
                            "ib .. .. sq sq .. .."
                            "ib .. .. sq sq .. .."
                            "ib .. .. .. .. .. .."
                            "ib jb jb jb jb jb jb";

    grid-template-rows: repeat(6, 3rem);
    grid-template-columns: repeat(7, 3rem);
}
/* 上記grid-template-areas:で指定している名前に当たる要素を用意してgrid-areaで同じ名前を付ける */
.ib {
    grid-area: ib;
    background-color: aqua;
}
.sq {
    grid-area: sq;
    background-color: yellow;
}
